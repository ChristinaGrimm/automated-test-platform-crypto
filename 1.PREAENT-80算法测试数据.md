def RoundFunctionConstraint(self, r):
    input_bits = self.input1_At_Round(r)         
    output_bits = self.output1_At_Round(r)       
    Sin_bits = self.S_in_At_Round(r)
    Sout_bits = self.S_out_At_Round(r)
    PermutationIn_bits = self.permutationIn_At_Round(r)
    PermutationOut_bits = self.permutationOut_At_Round(r)
    XORin1_bits = self.XOR_input1_At_Round(r)
    XORin2_bits = self.XOR_input2_At_Round(r)
    XORout_bits = self.XOR_output_At_Round(r)
    constraint = []
    constraint += self.equal(input_bits, Sin_bits)
    constraint += self.equal(Sout_bits, PermutationIn_bits)
    constraint += self.equal(PermutationOut_bits, XORin2_bits)
    constraint += self.equal(XORout_bits, output_bits)
    constraint += self.equal(output_bits, self.input1_At_Round(r + 1))
    constraint += self.Function_Sbox(Sin_bits, Sout_bits)
    constraint += self.Permutation(PermutationIn_bits, PermutationOut_bits)
    constraint += self.BVXOR(XORin1_bits, XORin2_bits, XORout_bits)
    constraint += ["\n"]
    return constraint



## 密钥生成算法：
def key_schedule_80(master_key: int) -> list:
    """
    master_key: 80-bit int
    return: list of 32 round keys (each 64-bit int)
    """
    assert 0 <= master_key < (1 << 80)
    K = master_key
    round_keys = []
    for round_counter in range(1, 33):  # 32 round keys
        # 1. 取最高 64 bit 作为 round key
        rk = (K >> 16) & ((1 << 64) - 1)
        round_keys.append(rk)

        # 2. 左循环 61 位（直接写在这里）
        K = ((K << 61) & ((1 << 80) - 1)) | (K >> 19)

        # 3. S 盒作用在最高 4 bit
        ms_nib = (K >> 76) & 0xF
        K = (K & ((1 << 76) - 1)) | (SBOX[ms_nib] << 76)

        # 4. 将轮计数器 (5 bit) 异或到 k19..k15
        rc = round_counter & 0x1F
        K ^= (rc << 15)

    return round_keys

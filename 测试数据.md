# 密钥生成算法
def generate_advanced_key(seed=None, length=32):
    """
    生成更复杂的密钥
    :param seed: 可选的种子值
    :param length: 密钥长度
    :return: base64编码的密钥
    """
    # 收集各种随机数据源
    random_data = secrets.token_bytes(64)
    timestamp = datetime.now().isoformat().encode()
    
    if seed:
        seed_data = str(seed).encode()
    else:
        seed_data = secrets.token_bytes(16)
    
    # 组合所有数据
    combined = random_data + timestamp + seed_data
    
    # 多次哈希增加复杂度
    hashed = hashlib.sha512(combined).digest()
    hashed = hashlib.sha256(hashed).digest()
    
    # 转换为base64并截取所需长度
    key_base64 = base64.urlsafe_b64encode(hashed).decode()
    return key_base64[:length]

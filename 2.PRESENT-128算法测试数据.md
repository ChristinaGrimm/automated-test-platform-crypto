## 轮函数：
def RoundFunctionConstraint(self, r):
    # 定义各阶段的变量
    input_bits = self.input_At_Round(r)           # 轮输入 (64 bits)
    output_bits = self.output_At_Round(r)         # 轮输出 (64 bits)
    round_key_bits = self.round_key_At_Round(r)   # 轮密钥 (64 bits)
    
    # 中间变量
    after_key_xor_bits = self.after_key_xor_At_Round(r)  # AddRoundKey后 (64 bits)
    sbox_in_bits = self.sbox_in_At_Round(r)              # S盒输入 (64 bits = 16×4 bits)
    sbox_out_bits = self.sbox_out_At_Round(r)            # S盒输出 (64 bits = 16×4 bits)
    perm_in_bits = self.perm_in_At_Round(r)              # 置换输入 (64 bits)
    perm_out_bits = self.perm_out_At_Round(r)            # 置换输出 (64 bits)
    
    constraint = []
    
    # 1. AddRoundKey: 输入与轮密钥异或
    constraint += self.equal(input_bits, after_key_xor_bits)  # 输入直接连接到异或输入1
    constraint += self.BVXOR(after_key_xor_bits, round_key_bits, sbox_in_bits)
    
    # 2. S-box层: 16个并行4-bit S盒
    # 将64位输入分成16个4-bit块
    for i in range(16):
        sbox_input_segment = sbox_in_bits[i*4 : (i+1)*4]
        sbox_output_segment = sbox_out_bits[i*4 : (i+1)*4]
        constraint += self.Function_Sbox(sbox_input_segment, sbox_output_segment)
    
    # 3. P-layer: 比特置换
    constraint += self.equal(sbox_out_bits, perm_in_bits)
    constraint += self.Permutation(perm_in_bits, perm_out_bits)
    
    # 置换输出即为本轮输出
    constraint += self.equal(perm_out_bits, output_bits)
    
    # 连接下一轮的输入（如果需要）
    if r < self.total_rounds:
        constraint += self.equal(output_bits, self.input_At_Round(r + 1))
    
    constraint += ['\n']
    return constraint

## 密钥生成算法：
def key_schedule_128(master_key: int) -> list:
    """
    master_key: 128-bit integer (0 <= master_key < 2**128), represented k127..k0
    returns: list of 32 round keys (each 64-bit int), used in order
    """
    assert 0 <= master_key < (1 << 128)
    K = master_key
    round_keys = []
    for round_counter in range(1, 33):  # generate 32 round keys (1..32)
        # extract round key = leftmost 64 bits (k127..k64)
        rk = (K >> 64) & ((1 << 64) - 1)
        round_keys.append(rk)

        # update key register (performed after extracting rk)
        # 1 rotate left by 61
        K = rol128(K, 61)

        # 2 apply S-box to left-most eight bits (i.e., bits 127..120)
        top_byte = (K >> 120) & 0xFF
        ms_nib = (top_byte >> 4) & 0xF
        ls_nib = top_byte & 0xF
        new_top_byte = (SBOX[ms_nib] << 4) | SBOX[ls_nib]
        # clear old top byte and set new one
        K = (K & ((1 << 120) - 1)) | (new_top_byte << 120)

        # 3 XOR round_counter (5 bits) into bits k66..k62 (LSB of rc maps to k62)
        rc = round_counter & 0x1F
        K ^= (rc << 62)

    return round_keys

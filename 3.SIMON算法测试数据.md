def RoundFunctionConstraint(self, r):
    # 获取当前轮的输入输出
    input1_bits = self.input1_At_Round(r)
    input2_bits = self.input2_At_Round(r)
    output1_bits = self.output1_At_Round(r)
    output2_bits = self.output2_At_Round(r)

    # 获取S盒输入输出
    S_in_bits = self.S_in_At_Round(r)
    S_out_bits = self.S_out_At_Round(r)

    # 获取置换输入输出
    PermutationIn_bits = self.permutation_input_At_Round(r)
    PermutationOut_bits = self.permutation_output_At_Round(r)

    # --------- S盒约束 ----------
    # 假设self.SBox_Constraint定义了S盒输入输出约束
    self.SBox_Constraint(S_in_bits, S_out_bits)

    # --------- 模加约束 ----------
    # 假设self.ModAdd_Constraint定义了模加约束
    self.ModAdd_Constraint(input1_bits, input2_bits, output1_bits)

    # --------- 异或约束 ----------
    # 假设self.Xor_Constraint定义了异或约束
    self.Xor_Constraint(input1_bits, input2_bits, output2_bits)

    # --------- 比特置换约束 ----------
    # 假设self.Permutation_Constraint定义了比特置换约束
    self.Permutation_Constraint(PermutationIn_bits, PermutationOut_bits)


def RoundFunctionConstraint(self, r):
    input1_bits = self.input1_At_Round(r)
    input2_bits = self.input2_At_Round(r)
    output1_bits = self.output1_At_Round(r)
    output2_bits = self.output2_At_Round(r)
    S_in_bits = self.S_in_At_Round(r)
    S_out_bits = self.S_out_At_Round(r)
    PermutationIn_bits = self.permutation_input_At_Round(r)
    PermutationOut_bits = self.permutation_output_At_Round(r)
    constraint = []
    constraint += self.SBox_Constraint(S_in_bits, S_out_bits)
    constraint += self.ModAdd_Constraint(input1_bits, input2_bits, output1_bits)
    constraint += self.Xor_Constraint(input1_bits, input2_bits, output2_bits)
    constraint += self.Permutation_Constraint(PermutationIn_bits, PermutationOut_bits)
    constraint += ['\n']
    
    return constraint

def key_schedule(master_key):
    # master_key: 64-bit整数
    k = []
    # 分块32-bit
    k.append(master_key & WORD_MASK)            # k0 低32位
    k.append((master_key >> 32) & WORD_MASK)    # k1 高32位

    for i in range(ROUNDS - m):
        tmp = ror(k[-1], 3, 32) ^ k[-m]        # ROR3(k_{i+1}) XOR k_i
        tmp ^= ror(k[-1], 1, 32)               # XOR ROR1(k_{i+1})
        tmp ^= 0xFFFFFFFC                        # c 常量 32-bit (2^32 - 4)
        tmp ^= z0[i % 62]                       # z序列
        tmp &= WORD_MASK
        k.append(tmp)
    return k[:ROUNDS]
